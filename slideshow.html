    Tail recursion

    Scala

    import scala.annotation.tailrec
@tailrec def loop: Unit = loop

def streamcopy(in: InputStream, out: OutputStream, buf: Array[Byte]) {
    @tailrec def read(len: Int): Unit = if (len > 0) {
        out.write(buf, 0, len)
        read(in.read(buf))
    }
    read(in.read(buf))
}

    Pattern Matching

    Scala

    case class Witch(name: String)
val KnightWhoSayRE = """^Knights Who Say (.*)""".r
def findFromTheMovie(a: Any) = a match {
    case "Bedemir" | "Galahad" | "Lancelot" | "Robin" => 
        "Found a Knight"
    case Witch(name) => 
        "Found a Witch named " + name
    case KnightWhoSayRE(what) => 
        "Found the Knights Who Say " + what
    case i: Int if i > 0 => 
        "Found the speed of a swallow"
    case <montyPython>{sirs @ _*}</montyPython> => 
        "Found a montyPython XML"
    case _ => 
        "Found nothing"
}

findFromTheMovie("Robin")
findFromTheMovie(Witch("a witch"))
